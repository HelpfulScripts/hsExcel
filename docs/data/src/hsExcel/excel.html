  45/**

  46 * reads and returns a promise for an {@link #/hsLog/hsNode.excelFile excel file}.

  47 * ```

  48 * {

  49 *     {@link excel.File.getSheetNames getSheetNames},

  50 *  {@link excel.File.getTableColumns getTableColumns},

  51 *  {@link excel.File.getRowsForTable getRowsForTable},

  52 *  {@link excel.File.getTable getTable},

  53 *  {@link excel.File.nextExcelColIndex nextExcelColIndex},

  54 *  {@link excel.File.getCellValue getCellValue}

  55 * }

  56 * ```

  57 * # Usage

  58 * ```

  59 * const excel = require('./hsNode.excel');

  60 * const excelFile = excel.excelFile('./aFile.xlsx');

  61 * ``` 

  62 * @param name the name of the Excel file to read

  63 * @returns an object of functions providing access to the contents of the excel file.

  64 */

  65export class File implements ExcelFile { 

  66 private wb:WorkBook;

  67

  68    constructor(name:string, options?:any) {

  69        this.wb = XLSX.readFile(name, options);

  70    }

  71

  72    public get workbook() { return this.wb; }

  74 /**

  75  * retrieves sheet names from a file

  76  * @returns {[string]} an array of sheet names

  77  */

  78 public getSheetNames():string[] {

  79 return this.wb.SheetNames;

  80 }

  82 /**

  83  * getTableColumns retrieves an array of consecutive valid column names.

  84  * @param sheetName the sheet name to retrieve cells from

  85  * @param startCol the first column of the table; defaults to 'A'.

  86  * @param row the row to iterate over; defaults to 1.

  87  * @returns an excel table description

  88  */

  89 public getTableColumns(sheetName:string, startCol='A', row=1):TableStruct {

  90        const sheet:WorkSheet = this.wb.Sheets[sheetName];

  91 return this.constructCol(sheetName, row, this.getConsecutiveColumnNames(sheet, row, startCol));

  92 }

  93

  94 /**

  95  * getRowsForTable returns a 2D array[r][c] of row values, where the columns match the provided 

  96  * columns names. 

  97  * @param table an array of column descriptors. 

  98  * @param maxRows if specified, determines the maximum number of rows to scan for. 

  99  * If omitted, iteration stops when the first row of empty values is encountered.

 100  */

 101 public getRowsForTable(table:TableStruct, maxRows=0):DataRow[] {

 102 if (!table.sheetName) { throw new Error('illegal table parameter in getRowsForTable'); }

 103 const sheet:WorkSheet = this.wb.Sheets[table.sheetName];

 104 const result:DataRow[] = [];

 105 let row=0; 

 106 while (true) {

 107 let newRow = this.getRow(sheet, row+table.headerRow+1, table.colIndex);

 108 let filledCells = 0;

 109 for (let c in newRow) { if (newRow[c]) { filledCells++; }}

 110            row++;

 111            // only return non-empty rows

 112            if (filledCells > 0) { result.push(newRow); }

 113            // if no maxRows specified: break upon first empty row

 114            else if (maxRows<=0) {  break; }

 115            // if rows exceed maxRows: break;

 116 if (maxRows>0 && row>=maxRows) { break; }

 117 }

 118 return result;

 119 }

 121    /**

 122     * **generator** for Excel column indices starting at startCol. 

 123     * Following 'Z' the next column generated is 'AA' and so on. The generator 

 124     * starts with producing startCol as first index.

 125     * # Usage

 126     * ```

 127     * for (col of file.nextExcelColIndex(startCol='Y') { 

 128     *    printf("%s, ", col);      // -> Y, Z, AA, AB

 129     *    if (col === 'AB')  { break; }

 130     * }

 131     * ```

 132     * @param startCol the first column index ('A', ....) to yield; defaults to 'A'

 133     */

 134    public* nextExcelColIndex(startCol='A'):IterableIterator {

 135        function nextChar(c:string):string { return String.fromCharCode(c.charCodeAt(0) + 1); }

 136        

 137        let c = startCol;

 138        while (true) {

 139            yield c;

 140            if (c.length === 1) {

 141                c = (c < 'Z')? nextChar(c[0]) : 'AA';

 142            } else {

 143                var ch = nextChar(c[1]);

 144                c = (ch > 'Z')? nextChar(c[0])+'A' : c[0] + ch;

 145            }

 146        }

 147    }

 149 /**

 150  * gets a table of values, starting at the startCol and startRow.

 151  * The table includes all consecutive columns with valid names, and all consecutive

 152  * rows with at least one valid cell value.

 153  * @param sheetID the sheet name or index from which to get the table

 154  * @param startCol determines the left edge of the table; defaults to 'A'

 155  * @param startRow determines the top edge of the table; defaults to 1

 156  * @returns a tuple of {columns, table} 

 157  */

 158 public getTable(sheetID:string|number, startCol='A', startRow=1) {

 159        const sheetName = (typeof sheetID === 'string')? sheetID : this.getSheetNames()[sheetID];

 160 const columns:TableStruct = this.getTableColumns(sheetName, startCol, startRow);

 161 const table:DataRow[]     = this.getRowsForTable(columns);

 162 return {columns, table};

 163    }

 165    /**

 166  * returns the value of a cell, or undefined

 167  * @param sheet the sheet object or sheet name to retrieve cells from;

 168  * @param col the column index ('A',...)

 169  * @param row index (1,...)

 170  * @returns the value of a cell, or undefined

 171  */

 172 public getCellValue(sheet:string|WorkSheet, col:string, row:number):string {

 173 if (typeof sheet === 'string') { sheet = this.wb.Sheets[sheet]; } 

 174        let c:CellObject;

 175 if (sheet[col+row] && sheet[col+row].v!=='') { 

 176            c = sheet[col+row];

 177            let val = c.w!==undefined? c.w : c.v;

 178 if (c) { switch(c.t) {

 179 case 's': return (val).replace(/,/g,';').replace(/[\n\r]+/g,' ').trim();

 180 case 'n': /* falls through */ 

 181 default: return c.w.replace(/,/g,'');

 182 }}

 183 }

 184 return ''; 

 185 }

 189 /**

 190  * **Generator**, yields consecutive cell values over a row

 191  * @param sheet the sheet object or sheet name to retrieve cells from;

 192  * @param row the row to iterate over

 193  * @param colIterator iterable over columns;

 194  * or an iterable that generates column indices.

 195  */

 196 private* getCellValues(sheet:WorkSheet, row:number, colIterator:string[]) {

 197 for (let col of colIterator) {

 198 yield this.getCellValue(sheet, col, row); 

 199 }

 200 }

 201

 202 /**

 203  * **Generator**, yields consecutive column names as an 

 204  * {col, name} object. 

 205  * The generator exits when the first empty column name is encountered.

 206  * @param sheet the sheet to scan

 207  * @param row the row to scan

 208  * @param startCol defaults to 'A'

 209  */

 210 private* getConsecutiveColumnNames(sheet:WorkSheet, row:number, startCol:string) {

 211 for (let col of this.nextExcelColIndex(startCol)) {

 212 if (!this.getCellValue(sheet, col, row)) { break; }

 213 yield {col:col, name:this.getCellValue(sheet, col, row)}; 

 214 }

 215 }

 216

 217 /**

 218  * returns an array[c] of values from columns that match indices provided `columns`.

 219  * @param sheet the sheet object or sheet name to retrieve cells from;

 220  * @param row the row to iterate over

 221  * @param columns a) an array of column names. b) an {from:'A', to:'Z'} object 

 222  * @return array of column values in the row

 223  */

 224 private getRow(sheet:WorkSheet, row:number, columns:string[]) {

 225 const result = [...this.getCellValues(sheet, row, columns)];

 226 return result;

 227 }

 228

 229 /**

 230  * returns the value of a cell, or undefined

 231  * @param sheetName the sheet object or sheet name to retrieve cells from;

 232  * @param row index (1,...)

 233     * @param it an iterator over columns

 234  * @returns the value of a cell, or undefined

 235  */

 236 private constructCol(sheetName:string, row:number, it:any):TableStruct {

 237 const result:TableStruct = {

 238 names:[],

 239 sheetName: sheetName,

 240 headerRow: row,

 241 colIndex:  []

 242 };

 243 for (let col of it) {

 244 result.names.push(col.name);

 245 result.colIndex.push(col.col);

 246 }

 247 return result;

 248 }
